vault policy write spring_boot_policy - <<EOF
path "secrets/kv_v2/*" {
  capabilities = ["read", "list", "create", "update", "delete" ]
}
EOF


vault write auth/approle/role/spring_boot_approle \
    token_policies="spring_boot_policy" \
    token_ttl=10s \
    token_max_ttl=10s

vault read auth/approle/role/spring_boot_approle/role-id

vault write -f auth/approle/role/spring_boot_approle/secret-id
